CLASS MeterReading
 ATTRIBUTES
  PRIVATE startReading : NUMBER
  PRIVATE endReading : NUMBER

CONSTRUCTOR(Start, End)
 SET startReading TO Start
 SET endReading TO End
END CONSTRUCTOR

METHODS
 GET_CONSUMPTION()
  RETURN endReading - startReading
 END METHOD
END CLASS



CLASS Bill
 ATTRIBUTES
  PRIVATE billingPeriod : TEXT
  PRIVATE dueDate : TEXT
  PRIVATE reading : MeterReading

CONSTRUCTOR(Period, DueDate, ReadingObj)
 SET billingPeriod TO Period
 SET dueDate TO DueDate
 SET reading TO ReadingObj
END CONSTRUCTOR

METHODS
 CALCULATE_AMOUNT_OWED()
  RETURN 0
 END METHOD

 GET_CONSUMPTION()
  RETURN reading.GET_CONSUMPTION()
 END METHOD

 SHOW_BILL()
  DISPLAY billingPeriod
  DISPLAY dueDate
 END METHOD
END CLASS

CLASS StandardRide INHERITS Ride
 ATTRIBUTES
  PRIVATE distance : NUMBER
  PRIVATE rate : NUMBER

CONSTRUCTOR(Pickup, Dropoff, RouteObj, Distance, Rate)
 CALL SUPER(Pickup, Dropoff, RouteObj)
 SET distance TO Distance
 SET rate TO Rate
END CONSTRUCTOR

METHODS
 CALCULATE_FARE()
  RETURN distance * rate
 END METHOD
END CLASS

CLASS PremiumRide INHERITS Ride
 ATTRIBUTES
  PRIVATE distance : NUMBER
  PRIVATE rate : NUMBER
  PRIVATE comfortSurcharge : NUMBER

CONSTRUCTOR(Pickup, Dropoff, RouteObj, Distance, Rate, Surcharge)
 CALL SUPER(Pickup, Dropoff, RouteObj)
 SET distance TO Distance
 SET rate TO Rate
 SET comfortSurcharge TO Surcharge
END CONSTRUCTOR

METHODS
 CALCULATE_FARE()
  RETURN (distance * rate) + comfortSurcharge
 END METHOD
END CLASS

CLASS SharedRide INHERITS StandardRide
 ATTRIBUTES
  PRIVATE passengers : NUMBER

CONSTRUCTOR(Pickup, Dropoff, RouteObj, Distance, Rate, Passengers)
 CALL SUPER(Pickup, Dropoff, RouteObj, Distance, Rate)
 SET passengers TO Passengers
END CONSTRUCTOR

METHODS
 CALCULATE_FARE()
  RETURN SUPER.CALCULATE_FARE() / passengers
 END METHOD
END CLASS
