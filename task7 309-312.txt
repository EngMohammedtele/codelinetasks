CLASS RollerCoaster INHERITS Ride
 ATTRIBUTES
  PRIVATE peopleInLine : NUMBER
  PRIVATE carCapacity : NUMBER
  PRIVATE cycleTime : NUMBER

CONSTRUCTOR(Name, Location, LogObj, People, Capacity, Cycle)
 CALL SUPER(Name, Location, LogObj)
 SET peopleInLine TO People
 SET carCapacity TO Capacity
 SET cycleTime TO Cycle
END CONSTRUCTOR

METHODS
 CALCULATE_WAIT_TIME()
  RETURN (peopleInLine / carCapacity) * cycleTime
 END METHOD
END CLASS
