PUBLIC

    SET claimID TO id
    SET amount TO amount
    SET status TO "Filed"

METHOD advanceStatus()
    IF status = "Filed" THEN
        SET status TO "Under Review"
    ELSE IF status = "Under Review" THEN

CLASS Policy
PRIVATE
    policyID : NUMBER
    policyType : TEXT
    premium : NUMBER
    coverages : LIST<Coverage>
    claims : LIST<Claim>

PUBLIC
CONSTRUCTOR(id,type,premium)
    SET policyID TO id
    SET policyType TO type
    SET premium TO premium
    CREATE EMPTY LIST coverages
    CREATE EMPTY LIST claims
END CONSTRUCTOR

METHOD addCoverage(coverage)
    ADD coverage TO coverages
END METHOD

METHOD addClaim(claim)
    ADD claim TO claims
END METHOD

METHOD validateClaim(claim,coverage)
    IF claim.amount <= coverage.limit THEN
        DISPLAY "Claim Accepted"
    ELSE
        DISPLAY "Claim Exceeds Coverage Limit"
    END IF
END METHOD

METHOD calculateRemainingCoverage(coverage,claim)
    DISPLAY coverage.limit - claim.amount
END METHOD

METHOD SHOW_POLICY()
    DISPLAY policyType
END METHOD
END CLASS


START

    CREATE OBJECT coverage1 OF Coverage(20,"Medical",1000)
    CREATE OBJECT claim1 OF Claim(100,500)
    CALL policy1.addCoverage(coverage1)
    CALL policy1.addClaim(claim1)
    CALL policy1.validateClaim(claim1,coverage1)
    CALL policy1.calculateRemainingCoverage(coverage1,claim1)
    CALL claim1.advanceStatus()
END
STOP

    DISPLAY workerName
END METHOD
END CLASS


CLASS Farm
PRIVATE
    farmID : NUMBER
    farmName : TEXT
    location : TEXT
    fields : LIST<Field>

PUBLIC
CONSTRUCTOR(id,name,location)
    SET farmID TO id
    SET farmName TO name
    SET location TO location
    CREATE EMPTY LIST fields
END CONSTRUCTOR

METHOD addField(field)
    ADD field TO fields
END METHOD

METHOD SHOW_FARM()
    DISPLAY farmName
END METHOD
END CLASS

    CREATE OBJECT activity1 OF Activity(1,"Watering",60)
    CREATE OBJECT worker1 OF Worker(50,"Ali","Farmer",5)
    CREATE OBJECT farm1 OF Farm(1,"Green Farm","Salalah")

    CALL farm1.addField(field1)
    CALL field1.logWateringActivity(activity1)
    CALL worker1.assignActivity()

END
