CLASS Genre
 ATTRIBUTES
  PRIVATE genreID : NUMBER
  PRIVATE genreName : TEXT
  PRIVATE description : TEXT
  PRIVATE books : LIST<Book>

CONSTRUCTOR(ID, Name, Description)

 SET genreName TO Name
 SET description TO Description
 SET books TO EMPTY LIST
END CONSTRUCTOR

METHODS
 ADD_BOOK(book : Book)

 END METHOD

  RETURN genreName

END CLASS

CLASS Attempt
 ATTRIBUTES
  PRIVATE attemptID : NUMBER
  PRIVATE exam : Exam
  PRIVATE student : Student
  PRIVATE answers : LIST<TEXT>
  PRIVATE score : NUMBER
  PRIVATE startTime : NUMBER

CONSTRUCTOR(ID, ExamObj, StudentObj, Start)
 SET attemptID TO ID
 SET exam TO ExamObj
 SET student TO StudentObj
 SET answers TO EMPTY LIST
 SET score TO 0
 SET startTime TO Start
END CONSTRUCTOR

METHODS
 ADD_ANSWER(answerText, CurrentTime)

  IF CurrentTime - startTime > exam.GET_DURATION() THEN
   DISPLAY "Time Expired"
  ELSE
   ADD answerText TO answers
  END IF

 END METHOD


 CALCULATE_SCORE()

  SET total TO 0
  SET questionList TO exam.GET_QUESTIONS()

  FOR i = 1 TO SIZE OF questionList

   IF answers[i] = questionList[i].GET_CORRECT_ANSWER() THEN
    SET total TO total + questionList[i].GET_POINTS()
   END IF

  END FOR

  SET score TO total
  RETURN score

 END METHOD


 SHOW_ATTEMPT()
  DISPLAY score
 END METHOD
END CLASS

 CREATE OBJECT exam1 OF Exam(1,"OOP Exam",60)
 CREATE OBJECT question1 OF Question(1,"What is OOP?","Object-Oriented Programming",10)
 CALL exam1.ADD_QUESTION(question1)
 CREATE OBJECT student1 OF Student(100,"Ahmed")
 CREATE OBJECT attempt1 OF Attempt(50,exam1,student1,0)
 student1.ADD_ATTEMPT(attempt1)
 attempt1.ADD_ANSWER("Object-Oriented Programming",20)
 DISPLAY attempt1.CALCULATE_SCORE()

END

CLASS Member
 ATTRIBUTES
  PRIVATE memberID : NUMBER
  PRIVATE memberName : TEXT
  PRIVATE phone : TEXT
  PRIVATE plan : MembershipPlan
  PRIVATE bookings : LIST<Booking>

CONSTRUCTOR(ID, Name, Phone, PlanObj)
 SET memberID TO ID
 SET memberName TO Name
 SET phone TO Phone
 SET plan TO PlanObj
 SET bookings TO EMPTY LIST
END CONSTRUCTOR

METHODS
 ADD_BOOKING(b : Booking)
  ADD b TO bookings
 END METHOD

 CALCULATE_USED_HOURS()
  SET total TO 0

  FOR EACH booking IN bookings
   SET total TO total + booking.GET_HOURS()
  END FOR

  RETURN total
 END METHOD

 GET_REMAINING_HOURS()
  RETURN plan.GET_QUOTA() - CALCULATE_USED_HOURS()
 END METHOD

 SHOW_MEMBER()
  DISPLAY memberName
 END METHOD
END CLASS                                                  

CLASS Room
 ATTRIBUTES
  PRIVATE roomID : NUMBER
  PRIVATE roomName : TEXT
  PRIVATE capacity : NUMBER
  PRIVATE bookings : LIST<Booking>

CONSTRUCTOR(ID, Name, Capacity)
 SET roomID TO ID
 SET roomName TO Name
 SET capacity TO Capacity
 SET bookings TO EMPTY LIST
END CONSTRUCTOR

METHODS
 ADD_BOOKING(b : Booking)
  ADD b TO bookings
 END METHOD

 IS_AVAILABLE(Date, Start, End)

  FOR EACH booking IN bookings

   IF booking.GET_DATE() = Date THEN

    IF Start < booking.GET_ENDTIME() AND End > booking.GET_STARTTIME() THEN
     RETURN FALSE
    END IF

   END IF

  END FOR

  RETURN TRUE

 END METHOD

 SHOW_ROOM()
  DISPLAY roomName
 END METHOD
END CLASS

CLASS Booking
 ATTRIBUTES
  PRIVATE bookingID : NUMBER
  PRIVATE date : TEXT
  PRIVATE startTime : NUMBER
  PRIVATE endTime : NUMBER
  PRIVATE hours : NUMBER
  PRIVATE member : Member
  PRIVATE room : Room

CONSTRUCTOR(ID, Date, Start, End, Hours, MemberObj, RoomObj)

 IF Hours > 0 AND Start < End THEN
  SET bookingID TO ID
  SET date TO Date
  SET startTime TO Start
  SET endTime TO End
  SET hours TO Hours
  SET member TO MemberObj
  SET room TO RoomObj
 ELSE
  DISPLAY "Invalid Booking"
 END IF

END CONSTRUCTOR

METHODS
 GET_HOURS()
  RETURN hours
 END METHOD

 GET_DATE()
  RETURN date
 END METHOD

 GET_STARTTIME()
  RETURN startTime
 END METHOD

 GET_ENDTIME()
  RETURN endTime
 END METHOD

 CONFIRM()

  IF member.GET_REMAINING_HOURS() < hours THEN
   DISPLAY "Quota Exceeded"

  ELSE IF room.IS_AVAILABLE(date,startTime,endTime) = FALSE THEN
   DISPLAY "Room Not Available"

  ELSE
   CALL member.ADD_BOOKING(this)
   CALL room.ADD_BOOKING(this)
   DISPLAY "Booking Confirmed"
  END IF

 END METHOD

 SHOW_BOOKING()
  DISPLAY date
 END METHOD
END CLASS

 CREATE OBJECT plan1 OF MembershipPlan(10,"Gold",20)

 CREATE OBJECT room1 OF Room(5,"Meeting Room A",10)
 CREATE OBJECT booking1 OF Booking(100,"2026-07-10",10,12,2,member1,room1)
 booking1.CONFIRM()
 DISPLAY member1.GET_REMAINING_HOURS()

END
STOP
